DELIMITER //

CREATE PROCEDURE getTitlesByActor(
    IN actor VARCHAR(30)
)

BEGIN
    SELECT original_title
    FROM Names_ AS N, Had_role AS H, Titles AS T
    WHERE N.name_ LIKE CONCAT('%', actor, '%')
    AND T.title_id = H.title_id
    AND N.name_id = H.name_id
    LIMIT 10;

END //

DELIMITER ;
